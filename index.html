            if (val >= 1000) {

                return (val / 1000).toFixed(1).replace(/\.0$/, '') + "B";

            }

            return val + "M";

        }

        // Calculation Logic

        function calculate() {

            // Energy Calculation

            const rawEnergy = (state.params * state.dutyCycle * ENERGY_CONST * state.pue); 

            

            const carbon = rawEnergy * CARBON_INTENSITY;

            const water = rawEnergy * WATER_INTENSITY;


            // Update KPIs

            els.kpiEnergy.innerText = rawEnergy.toFixed(1);

            els.kpiCarbon.innerText = carbon.toFixed(1);

            els.kpiWater.innerText = water.toFixed(1);


            // Update Footer Stats

            els.statMiles.innerText = Math.round(carbon * 2.56).toLocaleString();

            els.statPhones.innerText = Math.round(rawEnergy * 83.3).toLocaleString();

            els.statTrees.innerText = (carbon * 0.625).toFixed(1);


            // Update Impact Score (Normalized for view)

            // Carbon > 500 is high risk for this dashboard view

            let score = (carbon / 500) * 100;

            if (score > 99.9) score = 99.9;

            if (score < 1) score = 1;

            

            els.scoreValue.innerText = score.toFixed(1);


            // Update Score Styling

            const scoreEl = els.scoreValue;

            const labelEl = els.scoreLabel;

            const tagNist = els.nistTag;

            const tagEsg = els.esgTag;


            if (score > 75) {

                // High Risk

                scoreEl.className = "text-4xl font-black text-score-red";

                labelEl.className = "text-xs font-bold bg-red-100 text-red-600 px-1.5 py-0.5 rounded uppercase";

                labelEl.innerText = "High";

                tagNist.className = "text-[10px] font-bold bg-orange-100 text-orange-600 px-2 py-0.5 rounded uppercase";

                tagNist.innerText = "High Risk Tier";

                tagEsg.className = "text-[10px] font-bold bg-red-100 text-red-600 px-2 py-0.5 rounded uppercase";

                tagEsg.innerText = "Materiality Risk";

            } else if (score > 40) {

                // Medium Risk

                scoreEl.className = "text-4xl font-black text-score-orange";

                labelEl.className = "text-xs font-bold bg-orange-100 text-orange-600 px-1.5 py-0.5 rounded uppercase";

                labelEl.innerText = "Med";

                tagNist.className = "text-[10px] font-bold bg-amber-100 text-amber-600 px-2 py-0.5 rounded uppercase";

                tagNist.innerText = "Audit Needed";

                tagEsg.className = "text-[10px] font-bold bg-amber-100 text-amber-600 px-2 py-0.5 rounded uppercase";

                tagEsg.innerText = "Review";

            } else {

                // Low Risk

                scoreEl.className = "text-4xl font-black text-score-green";

                labelEl.className = "text-xs font-bold bg-green-100 text-green-600 px-1.5 py-0.5 rounded uppercase";

                labelEl.innerText = "Low";

                tagNist.className = "text-[10px] font-bold bg-green-100 text-green-600 px-2 py-0.5 rounded uppercase";

                tagNist.innerText = "Compliant";

                tagEsg.className = "text-[10px] font-bold bg-green-100 text-green-600 px-2 py-0.5 rounded uppercase";

                tagEsg.innerText = "Compliant";

            }


            // Update Charts

            if (barChart && radarChart) {

                // Update Bar Chart

                barChart.data.datasets[0].data = [rawEnergy, carbon, water];

                barChart.update();


                // Update Radar Chart (Normalized 0-100)

                const normEnergy = Math.min((rawEnergy / 1000) * 100, 100);

                const normCarbon = Math.min((carbon / 500) * 100, 100);

                const normWater = Math.min((water / 600) * 100, 100);

                // Scale normalized against 100B (100,000M)

                const normScale = Math.min((state.params / 10000) * 100, 100);


                radarChart.data.datasets[0].data = [normEnergy, normCarbon, normWater, normScale];

                radarChart.update();

            }

        }

        window.selectOption = function(cat, val, el) {

            // Set state

            wizardState[cat] = val;

            const siblings = el.parentElement.children;

            for (let i = 0; i < siblings.length; i++) {

        window.applyWizard = function() {

            // Defaults

            let newParams = 7000;

            let newDuty = 10000;


            // 1. Function Logic (Sets Base Scale)

            if (wizardState.func === 'imaging') newParams = 200; // CNN Scale

            if (wizardState.func === 'doc') newParams = 500; // BERT/NLP Scale

            if (wizardState.func === 'gen') newParams = 7000; // LLM Scale


            // 2. Type Logic (Multiplier)

            if (wizardState.type === 'foundation') {

                if (wizardState.func === 'imaging') newParams = 600; // e.g. SAM

                if (wizardState.func === 'doc') newParams = 7000; // e.g. Ambient LLM

                if (wizardState.func === 'gen') newParams = 70000; // e.g. GPT-4

            }


            // 3. Scale Logic (Sets Duty Cycle)

            if (wizardState.scale === 'dept') newDuty = 5000;

            if (wizardState.scale === 'ent') newDuty = 100000;


            // Apply

            state.params = newParams;

            state.dutyCycle = newDuty;


            // Update UI

            els.paramSlider.value = newParams;

            els.paramLabel.innerText = formatParams(newParams);

            els.dutySlider.value = newDuty;

            els.dutyLabel.innerText = newDuty.toLocaleString();

            els.modelSelect.value = 'Custom';


            // Close & Recalculate

            toggleModal('wizardModal');

            calculate();

        }


        // Event Listeners

        els.paramSlider.addEventListener('input', (e) => {

            state.params = parseInt(e.target.value);

            els.paramLabel.innerText = formatParams(state.params);

            calculate();

        });


        els.dutySlider.addEventListener('input', (e) => {

            state.dutyCycle = parseInt(e.target.value);

            els.dutyLabel.innerText = state.dutyCycle.toLocaleString();

            calculate();

        });


        els.modelSelect.addEventListener('change', (e) => {

            const model = e.target.value;

            // Presets based on approximate model sizes

            if (model === 'VizAI') {

                state.params = 200; // ~200M (CNN based)

            } else if (model === 'NuanceDAX') {

                state.params = 7000; // ~7B (Ambient)

            } else if (model === 'NuancePS') {

                state.params = 500; // ~500M (NLP)

            } else if (model === 'NuanceDragon') {

                state.params = 7000; // ~7B (DAX Copilot)

            } else if (model === 'Fluency') {

                state.params = 500; // ~500M (NLP/Speech)

            } else if (model === 'RadAI') {

                state.params = 7000; // ~7B (GenAI)

            }

            

            els.paramSlider.value = state.params;

            els.paramLabel.innerText = formatParams(state.params);

            calculate();

        });

            // Set active style

            btn.className = "pue-btn py-2 px-2 rounded-lg border border-transparent bg-indigo-600 text-white text-xs font-bold shadow-md ring-2 ring-indigo-100 transition-colors";

            

            calculate();

        }

            calculate();
